class Solution {
  public:
    void reverseArray(vector<int> &arr) {
        // code here
        int n=size(arr);
        int str=0;
        int end=n-1;
        while (str<end){
            int temp=arr[str];
            arr[str]=arr[end];
            arr[end]=temp;
        str++;
        end--;
        }
    }
};
